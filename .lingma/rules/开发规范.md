## 3. 错误处理标准

### 强制模式

```python
from core.langfuse_client import langfuse
from core.tracing import observe

class ObservabilityError(Exception):
    """可观测性错误的基类异常。"""
    pass

class TraceSubmissionError(ObservabilityError):
    """当向 Langfuse 提交追踪失败时抛出。"""
    pass

class ValidationError(Exception):
    """当输入验证失败时抛出。"""
    pass

@observe(name="process-request", as_type="span")
def process_request(request: dict) -> dict:
    """
    处理请求并进行适当的错误处理和追踪。

    所有异常会自动被捕获到追踪中。
    """
    try:
        # 验证输入
        if not request.get("session_id"):
            raise ValidationError("session_id 是必需的")

        # 处理请求
        result = perform_operation(request)

        return {"success": True, "data": result}

    except ValidationError as e:
        # 验证错误是预期的,记录 INFO 级别日志
        logger.info(f"验证失败: {e}")
        raise

    except TraceSubmissionError as e:
        # 可观测性错误不应导致操作失败
        logger.error(f"提交追踪失败: {e}")
        # 继续执行 - 优雅降级
        return perform_operation_without_tracing(request)

    except Exception as e:
        # 意外错误必须被捕获并重新抛出
        logger.exception(f"处理请求时发生意外错误: {e}")
        raise
```

### 规则

- ✅ **始终**在通用 `Exception` 之前捕获特定异常
- ✅ **始终**记录带有上下文的异常
- ✅ **始终**重新抛出异常,除非有意进行优雅降级
- ❌ **禁止**使用裸 `except:` 子句
- ❌ **禁止**静默抑制异常
- ❌ **禁止**捕获 `BaseException`(包括 KeyboardInterrupt)

### 脱敏规则

```python
def _mask_value_by_type(field_name: str, value: Any) -> Any:
    """根据字段类型脱敏值"""
    if not isinstance(value, str):
        return "***MASKED***"

    field_lower = field_name.lower()

    # 手机号脱敏: 138****5678
    if field_lower in ["phone", "mobile", "telephone"]:
        if len(value) >= 7:
            return value[:3] + "****" + value[-4:]
        return "***"

    # 邮箱脱敏: zh***@example.com
    if field_lower in ["email", "mail"]:
        parts = value.split("@")
        if len(parts) == 2 and len(parts[0]) > 2:
            return parts[0][:2] + "***@" + parts[1]
        return "***@***"

    # 身份证脱敏: 110101********1234
    if field_lower in ["id_card", "id_number", "identity"]:
        if len(value) >= 10:
            return value[:6] + "********" + value[-4:]
        return "***"

    # 姓名脱敏: 张*
    if field_lower in ["name", "username", "user_name"]:
        if len(value) > 1:
            return value[0] + "*" * (len(value) - 1)
        return "*"

    # 密码/密钥/令牌 - 完全隐藏
    if any(pattern in field_lower for pattern in ["password", "secret", "token", "key"]):
        return "***REMOVED***"

    return "***MASKED***"
```

### 用户 ID 哈希

```python
import hashlib

def hash_user_id(user_id: str) -> str:
    """
    为用户 ID 创建单向哈希以保护隐私,
    同时保持一致性以便追踪

    Args:
        user_id: 原始用户 ID

    Returns:
        哈希后的用户 ID (SHA-256,前16位)
    """
    return hashlib.sha256(user_id.encode()).hexdigest()[:16]
```

### 敏感字段列表

以下字段会被自动脱敏:
- `phone`, `mobile`, `telephone` - 电话号码
- `email`, `mail` - 邮箱地址
- `id_card`, `id_number`, `identity` - 身份证号
- `name`, `username`, `user_name` - 姓名
- `card_number`, `credit_card`, `bank_account` - 银行卡号
- `password`, `secret`, `token`, `api_key` - 密钥类
- `address`, `ip_address`, `ip` - 地址类

### 规则

- ✅ **始终**在发送到 Langfuse 前脱敏 PII (个人身份信息)
- ✅ **始终**对用户 ID 使用哈希而非明文
- ✅ **始终**使用环境变量存储 API 密钥
- ❌ **禁止**记录密码或 API 密钥
- ❌ **禁止**在日志中记录完整的请求/响应体

## 6. Langfuse 追踪规范

### Trace 层级设计

```
Trace (一次完整会话)
├── Span (意图识别)
│   └── Generation (LLM 调用)
```

### 装饰器使用

```python
from core.tracing import trace_customer_service, create_span, score_trace

@trace_customer_service(
    name="api_error_troubleshooting",
    session_id="session_123",
    user_id="user_456",
    tags=["technical_support", "api_issue"],
    metadata={"channel": "web_chat", "priority": "high"}
)
async def handle_api_error(query: str):
    """处理 API 错误的客服对话"""
    
    # 创建 Span
    with create_span("intent_recognition", input_data={"query": query}) as span:
        intent = await recognize_intent(query)
        span.end(output_data={"intent": intent, "confidence": 0.92})
    
    # 评分
    score_trace("intent_confidence", 0.92, comment="高置信度识别")
    
    return {"response": "解决方案...", "escalated": False}
```

### 评分体系

| 评分项 | 类型 | 范围 | 说明 |
|-------|------|------|------|
| `intent_confidence` | NUMERIC | 0-1 | 意图识别置信度 |
| `retrieval_relevance` | NUMERIC | 0-1 | RAG 检索相关性 |
| `tool_success_rate` | NUMERIC | 0-1 | 工具调用成功率 |
| `issue_resolved` | BOOLEAN | 0/1 | 问题是否解决 |
| `user_satisfaction` | NUMERIC | 1-5 | 用户满意度 |
| `response_latency_ms` | NUMERIC | ms | 响应延迟 |
| `failure_type` | CATEGORICAL | - | 失败类型分类 |
| `escalation_required` | BOOLEAN | 0/1 | 是否需要转接 |

```python
# ✅ 推荐: 带超时的异步 API 调用
async def fetch_with_timeout(

    max_retries: int = 3,
) -> Dict[str, Any]:
    """
    执行带超时和重试的 HTTP 请求。

    Args:
        url: 请求 URL
        timeout: 超时时间(秒)
        max_retries: 最大重试次数

    Returns:
        响应数据字典

    Raises:
        APITimeoutError: 请求超时
        APIRateLimitError: 速率限制
    """
    async with httpx.AsyncClient(timeout=timeout) as client:
        for attempt in range(max_retries):
            try:
                response = await client.get(url)
                response.raise_for_status()
                return response.json()
            except httpx.TimeoutException:
                if attempt == max_retries - 1:
                    raise APITimeoutError(f"Request timed out after {max_retries} attempts")
                wait_time = 2 ** attempt  # 指数退避
                await asyncio.sleep(wait_time)
            except httpx.HTTPStatusError as e:
                if e.response.status_code == 429:
                    retry_after = int(e.response.headers.get("Retry-After", 60))
                    await asyncio.sleep(retry_after)
                else:
                    raise
```

```python
from functools import lru_cache
from typing import Any, Dict, Optional
import time

# ✅ 推荐: 内存缓存装饰器
def cache_result(ttl: int = 300, max_size: int = 128):
    """
    带 TTL 的缓存装饰器。

    Args:
        ttl: 缓存存活时间(秒)
        max_size: 最大缓存条目数
    """
    def decorator(func):
        cache: Dict[str, tuple] = {}

        @functools.wraps(func)
        async def wrapper(*args, **kwargs):
            # 生成缓存键
            cache_key = f"{func.__name__}:{str(args)}:{str(sorted(kwargs.items()))}"
            
            # 检查缓存是否有效
            if cache_key in cache:
                result, timestamp = cache[cache_key]
                if time.time() - timestamp < ttl:
                    return result
                else:
                    del cache[cache_key]

            # 执行函数并缓存结果
            result = await func(*args, **kwargs)
            cache[cache_key] = (result, time.time())

            # 清理过期缓存
            if len(cache) > max_size:
                oldest_keys = sorted(cache.keys(), key=lambda k: cache[k][1])[:len(cache) - max_size]
                for key in oldest_keys:
                    del cache[key]

            return result

        return wrapper
    return decorator
```

### 9.4 API 客户端弹性设计

**强制模式**:

```python
from tenacity import (
    AsyncRetrying,
    retry_if_exception_type,
    stop_after_attempt,
    wait_exponential,
)
from pybreaker import CircuitBreaker
import httpx

# ✅ 推荐: 带熔断器和重试的 API 客户端
class ResilientAPIClient:
    """弹性 API 客户端,支持重试、熔断和限流"""

    def __init__(
        self,
        base_url: str,
        timeout: float = 30.0,
        max_retries: int = 3,
        circuit_breaker_threshold: int = 5,
        rate_limit_per_second: Optional[float] = None,
    ):
        self.base_url = base_url
        self.timeout = timeout
        self.max_retries = max_retries
        self.rate_limit = rate_limit_per_second
        
        # 初始化 HTTP 客户端
        self._client = httpx.AsyncClient(
            timeout=httpx.Timeout(timeout),
            headers={"Content-Type": "application/json"}
        )
        
        # 初始化熔断器
        self.circuit_breaker = CircuitBreaker(
            fail_max=circuit_breaker_threshold,
            reset_timeout=30  # 30秒后尝试恢复
        )
        
        # 限流控制
        self._last_request_time: Optional[float] = None

    async def request(self, method: str, endpoint: str, **kwargs) -> Dict:
        """
        执行带弹性的 HTTP 请求。

        Args:
            method: HTTP 方法
            endpoint: API 端点
            **kwargs: 请求参数

        Returns:
            响应数据
        """
        url = f"{self.base_url}/{endpoint.lstrip('/')}"
        
        # 检查熔断器状态
        if self.circuit_breaker.current_state == "open":
            raise CircuitBreakerOpenError("Service unavailable")
        
        # 应用限流
        await self._apply_rate_limit()
        
        # 执行带重试的请求
        async for attempt in AsyncRetrying(
            stop=stop_after_attempt(self.max_retries),
            wait=wait_exponential(multiplier=1, min=1, max=10),
            retry=retry_if_exception_type(
                (httpx.TimeoutException, httpx.NetworkError)
            ),
            reraise=True,
        ):
            try:
                response = await self._client.request(method, url, **kwargs)
                response.raise_for_status()
                
                # 记录成功
                self.circuit_breaker.call_succeed()
                return response.json()
                
            except Exception as e:
                self.circuit_breaker.call_fail()
                raise

    async def _apply_rate_limit(self):
        """应用速率限制"""
        if not self.rate_limit:
            return
        
        now = time.time()
        if self._last_request_time:
            elapsed = now - self._last_request_time
            min_interval = 1.0 / self.rate_limit
            
            if elapsed < min_interval:
                wait_time = min_interval - elapsed
                await asyncio.sleep(wait_time)
        
        self._last_request_time = time.time()
```

### 9.5 并发控制

**强制模式**:

```python
import asyncio
from typing import List, Any

# ✅ 推荐: 使用信号量限制并发
async def process_batch_with_concurrency(
    items: List[Any],
    process_func,
    max_concurrent: int = 10,
) -> List[Any]:
    """
    使用信号量控制并发处理批次数据。

    Args:
        items: 待处理项列表
        process_func: 处理函数
        max_concurrent: 最大并发数

    Returns:
        处理结果列表
    """
    semaphore = asyncio.Semaphore(max_concurrent)
    
    async def limited_process(item):
        async with semaphore:
            return await process_func(item)
    
    tasks = [limited_process(item) for item in items]
    results = await asyncio.gather(*tasks, return_exceptions=True)
    
    # 处理异常
    valid_results = []
    for result in results:
        if isinstance(result, Exception):
            logger.error(f"Task failed: {result}")
        else:
            valid_results.append(result)
    
    return valid_results

# 使用示例
```
